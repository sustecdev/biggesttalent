# File Upload Fix

; Add these to your php.ini or create a .htaccess file in the public directory

; Maximum file upload size (100MB)
upload_max_filesize = 100M

; Maximum POST data size (should be larger than upload_max_filesize)
post_max_size = 110M

; Maximum execution time (5 minutes for large uploads)
max_execution_time = 300

; Maximum input time
max_input_time = 300

; Memory limit
memory_limit = 256M
